例四、JS方式使用方法

1. 在页面上添加一个DIV

2. JS动态创建checkboxGroup

			function initData (obj) {
				var data = [{
					text: '足球',
					value: '0'
				},{
					text: '篮球',
					value: '1'
				},{
					text: '兵乓球',
					value: '2',
					readonly : "readonly"
				},{
					text: '羽毛球',
					value: '3'
				}]
				obj.setDatasource( data );
			}
			var a = cui('#checkboxGroup').checkboxGroup({
				name: 'like',
				checkbox_list: initData,
				direction: '',
				value: [0,1,2],
				readonly: false,
				on_change: function(value){
					alert('value change to ' + value);
				}
			});
