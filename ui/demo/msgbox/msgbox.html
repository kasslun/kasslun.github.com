一、自定义msgbox弹出位置

例： 
    /**
     * 自定义alert的位置
     */
    function openAlert(){
        cui.alert('自定义了位置', null, {
            position:'custom',
            customPos: {left: 20, top: 20}
        });
    }
    /**
     * 自定义message的位置
     */
    function openMessage(){
        cui.message('自定义了位置', 'success', {
            position:'custom',
            customPos: {left: 20, top: 20}
        });
    }
    /**
     * 自定义confirm的位置
     */
    function openConfirm(){
        cui.confirm('自定义了位置', {
            position:'custom',
            customPos: {left: 20, top: 20},
            onYes: function(){
                alert('接受了');
            },
            onNo: function(){
                alert('不接受');
            }
        });
    }
