1.2校验function

        function submitForm(isDisValid){
            if(isDisValid){
                //取消address的验证
                window.validater.disValid('address', true);
            }else{
                //恢复address的验证
                window.validater.disValid('address', false);
            }
            //验证表单
            window.validater.validAllElement();
        }
