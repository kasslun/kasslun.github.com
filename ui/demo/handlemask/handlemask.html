一、自定义的加载进度

    //自定义handleMask显示内容
    cui.handleMask({
        html: '
0%
'
    });

    var process = 0,
        intervalTime = null;
    //开始计算进度，这是一个模拟方法，原理是根据情况变更百分比
    function processHandle(){
        cui.handleMask.show();
        intervalTime = setInterval(function(){
            process ++;
            document.getElementById('process').innerHTML = process + '%';
            if(process === 100){
                cui.handleMask.hide();
                clearInterval(intervalTime);
            }
        }, 100);
    }
